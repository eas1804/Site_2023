Открыть несколько портов

iptables -A INPUT -p tcp -s 10.0.0.0/8 -m tcp -m multiport --ports 22,3006 -j ACCEPT -m comment --comment "LAN Calss A"
iptables -A INPUT -p tcp -s 172.16.0.0/12 -m tcp -m multiport --ports 22,3006 -j ACCEPT -m comment --comment "LAN Calss B"
iptables -A INPUT -p tcp -s 192.168.0.0/16 -m tcp -m multiport --ports 22,3006 -j ACCEPT -m comment --comment "LAN Calss C"
